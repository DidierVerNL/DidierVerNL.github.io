Random Generated Area's:

In ons spel ga je door verschillende area's, deze moesten ingespawned worden, maar ze moesten wel op een specifieke volgorde.


    [CreateAssetMenu(fileName = "Areas", menuName = "ScriptableObjects/AreaDatabase", order = 1)]
    public class AreaDatabase : ScriptableObject
    {
        [SerializeField] private AreaSO[] Areas;
    
        /// 
        /// Indexer
        /// 
        /// 
        public AreaSO this[int index] => Areas[index];
    
        public int AreaCount => Areas.Length;
    
        /// 
        /// Try Get, based on dictionary implementation
        /// 
        /// 
        public bool TryGetArea(string pName, out AreaSO pResult)
        {
            pResult = null;
            if (Areas == null) return false;
            if (Areas.Length <= 0) return false;
    
            for (int i = 0; i < Areas.Length; i++)
            {
                if (Areas[i].Name != pName) continue;
    
                pResult = Areas[i];
                return true;
            }
    
            return false;
        }
    
        /// 
        /// Try Get, based on dictionary implementation
        /// 
        /// 
        public bool TryGetArea(int pIndex, out AreaSO pResult)
        {
            pResult = null;
            if (Areas == null) return false;
            if (Areas.Length <= 0) return false;
    
            for (int i = 0; i < Areas.Length; i++)
            {
                if (Areas[i].Index != pIndex) continue;
    
                pResult = Areas[i];
                return true;
            }
    
            return false;
        }
    }
